<?php
session_start();
require_once 'config.php';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $table = $_POST['table'];
    unset($_POST['table'], $_POST['add']);
    
    $columns = implode(', ', array_keys($_POST));
    $placeholders = ':' . implode(', :', array_keys($_POST));
    
    $sql = "INSERT INTO $table ($columns) VALUES ($placeholders)";
    $stmt = $pdo->prepare($sql);
    
    if ($stmt->execute($_POST)) {
        $_SESSION['message'] = "تمت الإضافة بنجاح";
        header("Location: index.php?table=$table");
    } else {
        $_SESSION['error'] = "حدث خطأ في الإضافة";
    }
}

// عرض النموذج
$table = $_GET['table'] ?? $_POST['table'];
$stmt = $pdo->query("DESCRIBE $table");
$columns = $stmt->fetchAll(PDO::FETCH_COLUMN);

// إزالة الحقول التلقائية
$ignore = ['id', 'created_at'];
?>
<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <title>إضافة جديد</title>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.1.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>
    <div class="container mt-5">
        <h2>إضافة سجل جديد</h2>
        <form method="POST">
            <input type="hidden" name="table" value="<?php echo $table; ?>">
            
            <?php foreach($columns as $column): ?>
                <?php if (!in_array($column, $ignore)): ?>
                <div class="mb-3">
                    <label class="form-label"><?php echo $column; ?></label>
                    <input type="text" name="<?php echo $column; ?>" class="form-control" required>
                </div>
                <?php endif; ?>
            <?php endforeach; ?>
            
            <button type="submit" name="add" class="btn btn-primary">حفظ</button>
            <a href="index.php?table=<?php echo $table; ?>" class="btn btn-secondary">رجوع</a>
        </form>
    </div>
</body>
</html>
