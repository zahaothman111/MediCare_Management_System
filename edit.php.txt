<?php
session_start();
require_once 'config.php';

$table = $_GET['table'];
$id = $_GET['id'];

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    unset($_POST['update'], $_POST['table']);
    
    $setClause = [];
    foreach($_POST as $key => $value) {
        $setClause[] = "$key = :$key";
    }
    
    $sql = "UPDATE $table SET " . implode(', ', $setClause) . " WHERE id = :id";
    $stmt = $pdo->prepare($sql);
    $_POST['id'] = $id;
    
    if ($stmt->execute($_POST)) {
        $_SESSION['message'] = "تم التعديل بنجاح";
        header("Location: index.php?table=$table");
    }
}

// جلب البيانات الحالية
$stmt = $pdo->prepare("SELECT * FROM $table WHERE id = ?");
$stmt->execute([$id]);
$row = $stmt->fetch(PDO::FETCH_ASSOC);

// جلب أسماء الأعمدة
$stmt = $pdo->query("DESCRIBE $table");
$columns = $stmt->fetchAll(PDO::FETCH_COLUMN);
$ignore = ['id', 'created_at'];
?>
<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <title>تعديل بيانات</title>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.1.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>
    <div class="container mt-5">
        <h2>تعديل البيانات</h2>
        <form method="POST">
            <input type="hidden" name="table" value="<?php echo $table; ?>">
            
            <?php foreach($columns as $column): ?>
                <?php if (!in_array($column, $ignore)): ?>
                <div class="mb-3">
                    <label class="form-label"><?php echo $column; ?></label>
                    <input type="text" name="<?php echo $column; ?>" 
                           value="<?php echo htmlspecialchars($row[$column] ?? ''); ?>"
                           class="form-control">
                </div>
                <?php endif; ?>
            <?php endforeach; ?>
            
            <button type="submit" name="update" class="btn btn-primary">تحديث</button>
            <a href="index.php?table=<?php echo $table; ?>" class="btn btn-secondary">رجوع</a>
        </form>
    </div>
</body>
</html>
