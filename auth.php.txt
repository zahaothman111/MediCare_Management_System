<?php
session_start();

function isLoggedIn() {
    return isset($_SESSION['user_id']);
}

function hasPermission($requiredRole, $requiredPermission = null) {
    if ($_SESSION['role'] == 'super_admin') return true;
    if ($_SESSION['role'] == $requiredRole) return true;
    
    if ($requiredPermission && isset($_SESSION['permissions'])) {
        $permissions = json_decode($_SESSION['permissions'], true);
        return in_array($requiredPermission, $permissions);
    }
    
    return false;
}

function logActivity($action, $table = null, $record_id = null) {
    global $pdo;
    $stmt = $pdo->prepare("INSERT INTO activity_logs (user_id, action, table_name, record_id, ip_address) VALUES (?, ?, ?, ?, ?)");
    $stmt->execute([
        $_SESSION['user_id'],
        $action,
        $table,
        $record_id,
        $_SERVER['REMOTE_ADDR']
    ]);
}
?>
